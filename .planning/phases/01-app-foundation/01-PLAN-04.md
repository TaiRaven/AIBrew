**Step 1 — Create src/client/components/HomeView.tsx.**

Per UI-SPEC.md §7.2 and §5.2:
- 2-column grid on mobile (375px), 3-column on tablet (768px+), gap: --sp-lg (24px), padding: --sp-md (16px)
- Active tiles: Roasters, Equipment — `--aibrew-paper-2` background, 1px ink border at 20%, 8px radius, subtle -0.5deg tilt
- Disabled tiles: Beans, Brew Log, Recipes, History, Analytics — `--aibrew-disabled-bg` background, flat, `--aibrew-disabled-fg` text, no click handler
- Tile min-height: 96px

```tsx
// src/client/components/HomeView.tsx
import React from 'react'
import { navigateToView } from '../utils/navigate'

interface Tile {
  id: string
  label: string
  description: string
  active: boolean
  view?: string
  section?: string
}

const TILES: Tile[] = [
  { id: 'roasters',  label: 'Roasters',  description: 'Your roasters',      active: true,  view: 'catalog', section: 'roasters'  },
  { id: 'equipment', label: 'Equipment', description: 'Grinders & brewers', active: true,  view: 'catalog', section: 'equipment' },
  { id: 'beans',     label: 'Beans',     description: '',                   active: false },
  { id: 'brew',      label: 'Brew Log',  description: '',                   active: false },
  { id: 'recipes',   label: 'Recipes',   description: '',                   active: false },
  { id: 'history',   label: 'History',   description: '',                   active: false },
  { id: 'analytics', label: 'Analytics', description: '',                   active: false },
]

function TileCard({ tile }: { tile: Tile }) {
  const isActive = tile.active
  const tileIndex = TILES.indexOf(tile)
  const tilt = tileIndex % 2 === 0 ? 'rotate(-0.5deg)' : 'rotate(0.5deg)'

  const handleClick = () => {
    if (!isActive || !tile.view) return
    const params: Record<string, string> = {}
    if (tile.section) params.section = tile.section
    navigateToView(tile.view, params, `AIBrew — ${tile.label}`)
  }

  return (
    <div
      role="article"
      onClick={isActive ? handleClick : undefined}
      style={{
        minHeight: '96px',
        padding: 'var(--sp-md)',
        backgroundColor: isActive ? 'var(--aibrew-paper-2)' : 'var(--aibrew-disabled-bg)',
        border: isActive ? '1px solid rgba(29,26,23,0.2)' : 'none',
        borderRadius: '8px',
        transform: isActive ? tilt : 'none',
        cursor: isActive ? 'pointer' : 'default',
        display: 'flex',
        flexDirection: 'column',
        justifyContent: 'center',
        gap: 'var(--sp-xs)',
        boxSizing: 'border-box',
      }}
    >
      <div style={{
        fontFamily: 'var(--aibrew-font-disp)',
        fontSize: '20px',
        fontWeight: 600,
        lineHeight: 1.2,
        color: isActive ? 'var(--aibrew-ink)' : 'var(--aibrew-disabled-fg)',
      }}>
        {tile.label}
      </div>
      {isActive && tile.description && (
        <div style={{
          fontFamily: 'var(--aibrew-font-body)',
          fontSize: '16px',
          fontWeight: 400,
          lineHeight: 1.5,
          color: 'var(--aibrew-ink-3)',
        }}>
          {tile.description}
        </div>
      )}
    </div>
  )
}

export default function HomeView() {
  return (
    <div style={{ padding: 'var(--sp-md)', paddingTop: 'var(--sp-lg)' }}>
      <h1 style={{
        fontFamily: 'var(--aibrew-font-disp)',
        fontSize: '28px',
        fontWeight: 600,
        lineHeight: 1.2,
        color: 'var(--aibrew-ink)',
        margin: '0 0 var(--sp-lg) 0',
      }}>
        AIBrew
      </h1>
      <div style={{
        display: 'grid',
        gridTemplateColumns: 'repeat(2, 1fr)',
        gap: 'var(--sp-lg)',
      }}>
        {TILES.map(tile => <TileCard key={tile.id} tile={tile} />)}
      </div>
    </div>
  )
}
```

```tsx
// src/client/components/CatalogView.tsx
import React, { useEffect } from 'react'
import { getViewParams, navigateToView } from '../utils/navigate'

// RoasterSection and EquipmentSection are created in 01-PLAN-04 T02 and 01-PLAN-05
// Import them once created. Placeholders below until those plans complete.

// import EquipmentSection from './EquipmentSection'

const SUB_NAV_ITEMS = [
  { id: 'roasters',  label: 'Roasters',  disabled: false },
  { id: 'equipment', label: 'Equipment', disabled: false },
  { id: 'beans',     label: 'Beans',     disabled: true  },  // Phase 2
  { id: 'recipes',   label: 'Recipes',   disabled: true  },  // Phase 3
]

interface CatalogViewProps {
  params: URLSearchParams
}

export default function CatalogView({ params }: CatalogViewProps) {
  const section = params.get('section') || 'roasters'

  // Default to roasters section if no section param
  useEffect(() => {
    if (!params.get('section')) {
      navigateToView('catalog', { section: 'roasters' }, 'AIBrew — Catalog')
    }
  }, [])

  const handleSectionChange = (sectionId: string) => {
    const item = SUB_NAV_ITEMS.find(i => i.id === sectionId)
    if (!item || item.disabled) return
    navigateToView('catalog', { section: sectionId }, `AIBrew — ${item.label}`)
  }

  function renderSection() {
    switch (section) {
      case 'roasters':
        // Replace with <RoasterSection /> once 01-PLAN-04 T02 is complete
        return <div style={{ padding: 'var(--sp-md)', color: 'var(--aibrew-ink)', fontFamily: 'var(--aibrew-font-body)' }}>Roasters section (placeholder — replace with RoasterSection)</div>
      case 'equipment':
        // Replace with <EquipmentSection /> once 01-PLAN-05 is complete
        return <div style={{ padding: 'var(--sp-md)', color: 'var(--aibrew-ink)', fontFamily: 'var(--aibrew-font-body)' }}>Equipment section (placeholder — replace with EquipmentSection)</div>
      default:
        return null
    }
  }

  return (
    <div>
      {/* Catalog sub-navigation */}
      <div style={{
        display: 'flex',
        gap: 0,
        height: '40px',
        alignItems: 'center',
        backgroundColor: 'var(--aibrew-paper-2)',
        padding: '0 var(--sp-md)',
        borderBottom: '1px dashed rgba(158,136,116,0.4)',
      }}>
        {SUB_NAV_ITEMS.map((item, idx) => {
          const isActive = item.id === section
          const isDisabled = item.disabled
          return (
            <button
              key={item.id}
              onClick={() => handleSectionChange(item.id)}
              disabled={isDisabled}
              aria-disabled={isDisabled}
              style={{
                background: 'none',
                border: 'none',
                borderBottom: isActive ? '2px solid var(--aibrew-accent)' : '2px solid transparent',
                padding: '0 var(--sp-md)',
                height: '40px',
                fontFamily: 'var(--aibrew-font-body)',
                fontSize: '14px',
                fontWeight: 400,
                lineHeight: 1.4,
                color: isDisabled ? 'var(--aibrew-disabled-fg)' : isActive ? 'var(--aibrew-ink)' : 'var(--aibrew-ink-3)',
                cursor: isDisabled ? 'default' : 'pointer',
              }}
            >
              {item.label}
            </button>
          )
        })}
      </div>

      {/* Section content */}
      {renderSection()}
    </div>
  )
}
```

Note: The sub-nav uses `<button>` elements here because they are semantic nav elements managed entirely by CatalogView — not form inputs. // =============================================================================
```tsx
// DETAIL / EDIT VIEW
// =============================================================================
function RoasterDetailView({ sysId }: { sysId: string }) {
  const [showArchiveConfirm, setShowArchiveConfirm] = useState(false)
  const [archiveError, setArchiveError] = useState('')

  const handleBack = () => {
    navigateToView('catalog', { section: 'roasters' }, 'AIBrew — Roasters')
  }

  const handleArchiveConfirm = async () => {
    setArchiveError('')
    try {
      const response = await fetch(`/api/now/table/${ROASTER_TABLE}/${sysId}`, {
        method: 'PATCH',
        headers: {
          'Content-Type': 'application/json',
          'X-UserToken': (window as any).g_ck || '',
        },
        body: JSON.stringify({ active: 'false' }),
      })
      if (!response.ok) throw new Error('Archive failed')
      setShowArchiveConfirm(false)
      handleBack()
    } catch {
      setShowArchiveConfirm(false)
      setArchiveError('Couldn\'t archive — try again in a moment.')
    }
  }

  return (
    <div style={{ padding: 'var(--sp-md)' }}>
      <button
        onClick={handleBack}
        style={{
          background: 'none',
          border: 'none',
          color: 'var(--aibrew-accent)',
          fontFamily: 'var(--aibrew-font-body)',
          fontSize: '16px',
          cursor: 'pointer',
          padding: 0,
          marginBottom: 'var(--sp-md)',
          minHeight: '44px',
        }}
      >
        ← Back
      </button>
      {archiveError && (
        <div style={{ color: 'var(--aibrew-destructive)', fontFamily: 'var(--aibrew-font-body)', fontSize: '16px', marginBottom: 'var(--sp-sm)' }}>
          {archiveError}
        </div>
      )}
      <RecordProvider
        table={ROASTER_TABLE}
        sysId={sysId}
        isReadOnly={false}
      >
        <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', marginBottom: 'var(--sp-md)' }}>
          <FormActionBar
            onSubmit={handleBack}
            onCancel={handleBack}
          />
          <button
            onClick={() => setShowArchiveConfirm(true)}
            style={{
              backgroundColor: 'transparent',
              color: 'var(--aibrew-destructive)',
              border: '1px solid var(--aibrew-destructive)',
              borderRadius: '16px',
              padding: '4px 8px',
              fontFamily: 'var(--aibrew-font-body)',
              fontSize: '14px',
              cursor: 'pointer',
              minHeight: '32px',
            }}
          >
            Archive
          </button>
        </div>
        <FormColumnLayout />
      </RecordProvider>
      <ArchiveConfirmModal
        isOpen={showArchiveConfirm}
        onConfirm={handleArchiveConfirm}
        onCancel={() => setShowArchiveConfirm(false)}
      />
    </div>
  )
}
```
