function EquipmentDetailView({ sysId }: { sysId: string }) {
  // Same pattern as RoasterDetailView:
  // Back: NowButton variant="tertiary" → navigateToView('catalog', { section: 'equipment' })
  // RecordProvider + FormActionBar + FormColumnLayout
  // Archive chip: NowButton variant="secondary" (styled with destructive colour) → ArchiveEquipmentModal → PATCH active='false' → back on success
  // Error state: "Couldn't archive — try again in a moment."

function EquipmentListView({ onNew }: { onNew: () => void }) {
  const handleRowClick = (e: CustomEvent) => {
    const sysId = e.detail?.payload?.sys_id || e.detail?.sys_id
    if (!sysId) return
    navigateToView('catalog', { section: 'equipment', id: sysId }, 'AIBrew — Equipment')
  }

  return (
    <div style={{ padding: 'var(--sp-md)' }}>
      <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', marginBottom: 'var(--sp-lg)' }}>
        <h2 style={{
          fontFamily: 'var(--aibrew-font-disp)',
          fontSize: '28px',
          fontWeight: 600,
          color: 'var(--aibrew-ink)',
          margin: 0,
        }}>
          Equipment
        </h2>
        {/* Primary CTA — NowButton variant="primary", not raw <button> */}
        <NowButton
          onClick={onNew}
          variant="primary"
          style={{ minHeight: '44px' }}
        >
          New equipment
        </NowButton>
      </div>
      <NowRecordListConnected
        table={EQUIPMENT_TABLE}
        listTitle="Equipment"
        columns="name,type,notes"
        query="active=true"
        onRowClicked={handleRowClick}
        limit={50}
      />
    </div>
  )
}

export default function EquipmentSection() {
  const params = getViewParams()
  const sysId = params.get('id')
  const [showCreateModal, setShowCreateModal] = useState(false)

  if (sysId) {
    return <EquipmentDetailView sysId={sysId} />

      <EquipmentListView onNew={() => setShowCreateModal(true)} />
      <CreateEquipmentModal
        isOpen={showCreateModal}
        onClose={() => setShowCreateModal(false)}
      />
    </>
  )
}
